            <p className="text-muted-foreground">Security rules are a critical component of Firestore, providing declarative access control directly in the database to protect your data from unauthorized access.</p>
            <CodeBlock language="text">{`// In your firestore.rules file
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Only authenticated users can read any document in 'public_profiles'
    match /public_profiles/{userId} {
      allow read: if request.auth != null;
    }
    
    // Users can only create, read, update, or delete their own user document in 'users'
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Example for a 'posts' collection
    match /posts/{postId} {
      // Anyone can read posts
      allow read: if true;
      // Only authenticated users can create posts
      allow create: if request.auth != null;
      // Only the author of a post can update or delete it
      allow update, delete: if request.auth != null && resource.data.authorUid == request.auth.uid;
    }
  }
}
            `}</CodeBlock>

// functions/src/index.ts (example structure)

import * as functions from "firebase-functions";

try {
    admin.initializeApp();

// Trigger a function when a new user signs up
export const sendWelcomeEmailOnUserCreate = functions.auth.user().onCreate(async (user) => {
  const email = user.email; // The email of the user.
  const displayName = user.displayName || 'New User';
  
  if (!email) {
    console.error("User email is undefined. Cannot send welcome email.");
    return null;
  }
  // ... logic to send a welcome email (e.g., using SendGrid, Nodemailer)
  console.log(\`Sending welcome email to \${email} for user \${displayName}\`);
  
  // Example: Add a new document to Firestore when a user is created
  try {
    await admin.firestore().collection('users').doc(user.uid).set({
      email: email,
      displayName: displayName,
      createdAt: admin.firestore.FieldValue.serverTimestamp(),
    });
    console.log(\`User record created in Firestore for \${user.uid}\`);
  } catch (error) {
    console.error("Error creating user record in Firestore:", error);
  }
  return null;
});

// Trigger a function when a document is created in the 'orders' collection
export const processNewOrderInFirestore = functions.firestore
  .document('orders/{orderId}')
  .onCreate(async (snap, context) => {
    const newOrder = snap.data();
    const orderId = context.params.orderId;
    
    console.log(\`Processing new order: \${orderId}\`, newOrder);
    
    // Example: Update the order document with a 'processed' status
    try {
      await snap.ref.set({ processed: true, processedAt: admin.firestore.FieldValue.serverTimestamp() }, { merge: true });
      console.log(\`Order \${orderId} marked as processed.\`);
    } catch (error) {
      console.error(\`Error updating order \${orderId}:\`, error);
    }
    return null;
  });
            `}</CodeBlock>
